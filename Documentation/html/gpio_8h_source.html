313//
314// The values of the control bits were selected such that when only using 2 bits
315// open-drain outputs, pulled-up inputs, and inverted pins are still supported in
316// addition to the default push-pull outputs and floating inputs. If that's ever
317// takend advantage of the exported documentation should be updated to mention it.
318//
319// I'd like to use an enum here but that interferes with preprocessor tests
320// of pins when the control bits are used.
321/*
324typedef enum {
327 GPIO_CTRL_TRISTATE_HIGH = (0b010U << GPIO_CTRL_OFFSET),
330 GPIO_CTRL_TRISTATE_LOW = (0b100U << GPIO_CTRL_OFFSET),
333 GPIO_CTRL_BIAS_INPUT = (0b010U << GPIO_CTRL_OFFSET),
336 GPIO_CTRL_BIAS_LOW = (0b100U << GPIO_CTRL_OFFSET),
341 GPIO_CTRL_INVERT = (0b001U << GPIO_CTRL_OFFSET),
342} gpio_control_t;
343*/
346#define GPIO_CTRL_TRISTATE_HIGH (0b010U << GPIO_CTRL_OFFSET)
349#define GPIO_CTRL_TRISTATE_LOW (0b100U << GPIO_CTRL_OFFSET)
352#define GPIO_CTRL_BIAS_INPUT (0b010U << GPIO_CTRL_OFFSET)
355#define GPIO_CTRL_BIAS_LOW (0b100U << GPIO_CTRL_OFFSET)
358#define GPIO_CTRL_INVERT (0b001U << GPIO_CTRL_OFFSET)
361#define GPIO_CTRL_PULLUP (GPIO_CTRL_BIAS_INPUT)
364#define GPIO_CTRL_PULLDOWN (GPIO_CTRL_BIAS_INPUT|GPIO_CTRL_BIAS_LOW)
367#define GPIO_CTRL_FLOAT (0U)
370#define GPIO_CTRL_OPENDRAIN (GPIO_CTRL_TRISTATE_HIGH)
373#define GPIO_CTRL_PUSHPULL (0U)
